4.1 Javascript

text-mode

# key: function
# --
function(${1:args}) {
    $0
}

# name: var (declare)
# key: var
# --
var $0;

# name: var (assign)
# key: var
# --
var $1 = $0;

# name: var (assign)
# key: var
# --
describe("$1", function(done) {
        $0
});

# name: var (assign)
# key: var
# --

        $0
});
